/* Write a program to input marks of students n- student in computer sicene and calculate average and also counts number of students failed less than 40 */
#include<stdio.h>
int main()
{
    int n,computer_marks[1000],i,count=0,sum=0;
    float avg;
    printf("Enter total number of students to input marks:- ");
    scanf("%d",&n);
    for(i=1;i<=n;i++)
    {
       printf("Enter marks of Student %d :- ",i);
       scanf("%d",&computer_marks[i]);
       sum=sum+computer_marks[i];
       if (computer_marks[i]<40)
       {
           count=count+1;
       }
    }
    avg=sum/n;
    printf("The Average Marks of Student is %.2f\t\n",avg);
    printf("Total Number of Failed Students is %d\t ",count);
}
